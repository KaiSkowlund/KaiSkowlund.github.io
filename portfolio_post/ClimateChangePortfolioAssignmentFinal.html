# --- Imports ---
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.linear_model import LinearRegression
import pandas as pd

# --- Ensure temperature_c exists ---
if 'temperature_c' not in climate_df_c.columns:
    climate_df_c['temperature_c'] = (climate_df_c['temperature_f'] - 32) * (5/9)

# --- Ensure datetime index ---
if climate_df_c.index.name != 'DATE':
    climate_df_c['DATE'] = pd.to_datetime(climate_df_c['DATE'])
    climate_df_c = climate_df_c.set_index('DATE')

# --- Annual mean ---
ann_climate_df = climate_df_c['temperature_c'].resample('YE').mean().dropna()

# --- Prepare data for regression ---
X = ann_climate_df.index.year.values.reshape(-1, 1)
y = ann_climate_df.values
model = LinearRegression()
model.fit(X, y)
y_pred = model.predict(X)

# --- Plot ---
plt.figure(figsize=(12,6))
# Scatter of annual temps
sns.scatterplot(x=ann_climate_df.index.year, y=y, color='steelblue', s=50, label='Annual Temp (C)')
# Regression line
plt.plot(ann_climate_df.index.year, y_pred, color='darkred', linewidth=2.5, label='Linear Trend')
plt.title('Annual Mean Temperature Trend (C)', fontsize=16)
plt.xlabel('Year')
plt.ylabel('Temperature (C)')
plt.grid(True, alpha=0.3)
plt.legend()
plt.show()
